int cariNilaiTertinggi(int nilai[], int n) {
    int max = nilai[0];
    for (int i = 1; i < n; ++i) {
        if (nilai[i] > max) {
            max = nilai[i];
        }
    }
    return max;
}

int cariNilaiTerendah(int nilai[], int n) {
    int min = nilai[0];
    for (int i = 1; i < n; ++i) {
        if (nilai[i] < min) {
            min = nilai[i];
        }
    }
    return min;
}

float hitungRataRata(int nilai[], int n) {
    int total = 0;
    for (int i = 0; i < n; ++i) {
        total += nilai[i];
    }
    return static_cast<float>(total) / n;
}
